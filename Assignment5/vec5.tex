For getting the value of the $\angle OPR$
\begin{align}
    \cos{\angle OPR }&= \frac{\vec{\brak{O-P}}^{\top}\vec{\brak{R-P}}}{\vec{\norm{O-P}\norm{R-P}}}\\
    \angle OPR &= 10\degree
  \end{align}
